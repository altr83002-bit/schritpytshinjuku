-- WHITELIST
local Whitelist = {}
local function isWhitelisted(plr)
	return plr and Whitelist[plr.Name] == true
end
local whitelistHighlights = {}
local function applyWhitelistHighlight(plr)
	if not plr or not plr.Character then return end
	if whitelistHighlights[plr] then return end
	local h = Instance.new("Highlight")
	h.Name = "_WhitelistHL"
	h.Adornee = plr.Character
	h.FillTransparency = 1
	h.OutlineTransparency = 0
	h.OutlineColor = Color3.fromRGB(0,170,255)
	h.Parent = plr.Character
	whitelistHighlights[plr] = h
end
local function removeWhitelistHighlight(plr)
	if whitelistHighlights[plr] then
		whitelistHighlights[plr]:Destroy()
		whitelistHighlights[plr] = nil
	end
end
local function updateWhitelistForPlayer(plr)
	if isWhitelisted(plr) then
		applyWhitelistHighlight(plr)
	else
		removeWhitelistHighlight(plr)
	end
end

-- helpers
local function clearHighlight()
	if highlightTween then
		highlightTween:Cancel()
		highlightTween = nil
	end
	if currentHighlight then
		currentHighlight:Destroy()
		currentHighlight = nil
	end
end

-- client sided headless
local headlessEnabled = false
local function applyHeadless(char)
	local head = char:FindFirstChild("Head")
	if not head then return end
	head.Transparency = headlessEnabled and 1 or 0
	local face = head:FindFirstChildWhichIsA("Decal")
	if face then
		face.Transparency = headlessEnabled and 1 or 0
	end
end
local function applyCosmetics()
	if player.Character then
		applyHeadless(player.Character)
	end
end
player.CharacterAdded:Connect(function()
	lockedPart = nil
	clearHighlight()
	task.wait(0.2)
	applyCosmetics()
end)

-- under check
local function isTargetUnderLocalPlayer(char)
	local head = char:FindFirstChild("Head")
	local myChar = player.Character
	if not myChar then return false end
	local myRoot = myChar:FindFirstChild("HumanoidRootPart")
	if not head or not myRoot then return false end
	return head.Position.Y < myRoot.Position.Y
end

-- SKY PERSISTENCE
local function enforceSky()
	if not purpleSkyEnabled and not redDemonSkyEnabled and not heavenSkyEnabled then return end
	for _,obj in ipairs(Lighting:GetChildren()) do
		if obj:IsA("Sky") and obj.Name ~= "_v4mpSkyOnly" then
			obj:Destroy()
		end
	end
	if purpleSkyEnabled then
		setPurpleSky(true)
	elseif redDemonSkyEnabled then
		setRedDemonSky(true)
	elseif heavenSkyEnabled then
		setHeavenSky(true)
	end
end
Lighting.ChildAdded:Connect(function(c)
	if c:IsA("Sky") then
		task.defer(enforceSky)
	end
end)
Lighting.ChildRemoved:Connect(function(c)
	if c:IsA("Sky") then
		task.defer(enforceSky)
	end
end)

local MAX_BILLBOARDS = 300
-- Skip shop/display items
local function isShopItem(obj)
	local p = obj.Parent
	while p do
		local n = p.Name:lower()
		if n:find("shop") or n:find("vendor") or n:find("store") or n:find("display") or
		   n:find("case") or n:find("locker") or n:find("shelf") or n:find("rack") or
		   n:find("showcase") or n:find("cabinet") then
			return true
		end
		p = p.Parent
	end
	return false
end
local function getBestPart(obj)
	if obj.PrimaryPart then return obj.PrimaryPart end
	return obj:FindFirstChild("Handle") or obj:FindFirstChildWhichIsA("BasePart", true)
end
local function createBillboard(item, name, color)
	if not item:IsDescendantOf(Workspace) then return end
	if isShopItem(item) then return end
	if #billboards >= MAX_BILLBOARDS then return end
	local part = getBestPart(item)
	if not part then return end
	local existing = item:FindFirstChild("ItemESP")
	if existing then existing:Destroy() end
	local bb = Instance.new("BillboardGui")
	bb.Name = "ItemESP"
	bb.Adornee = part
	bb.Size = UDim2.new(0, 160, 0, 60)
	bb.StudsOffset = Vector3.new(0, 3.2, 0)
	bb.AlwaysOnTop = true
	bb.MaxDistance = 15000
	bb.Parent = item
	local nameLbl = Instance.new("TextLabel")
	nameLbl.Size = UDim2.new(1,0,0.6,0)
	nameLbl.BackgroundTransparency = 0.5
	nameLbl.BackgroundColor3 = Color3.new(0,0,0)
	nameLbl.Text = name
	nameLbl.TextColor3 = color
	nameLbl.TextScaled = true
	nameLbl.Font = Enum.Font.Oswald
	nameLbl.TextStrokeTransparency = 0.6
	nameLbl.TextStrokeColor3 = Color3.new(0,0,0)
	nameLbl.Parent = bb
	Instance.new("UICorner", nameLbl).CornerRadius = UDim.new(0,6)
	local distLbl = Instance.new("TextLabel")
	distLbl.Size = UDim2.new(1,0,0.4,0)
	distLbl.Position = UDim2.new(0,0,0.6,0)
	distLbl.BackgroundTransparency = 1
	distLbl.TextColor3 = Color3.new(1,1,1)
	distLbl.TextScaled = true
	distLbl.Font = Enum.Font.Gotham
	distLbl.TextStrokeTransparency = 0.5
	distLbl.TextStrokeColor3 = Color3.new(0,0,0)
	distLbl.Parent = bb
	billboards[item] = {bb = bb, dist = distLbl, part = part, enabled = espToggles[name] and true}
	bb.Enabled = billboards[item].enabled
end
local function tryTag(obj)
	if not (obj:IsA("Tool") or obj:IsA("Model")) then return end
	if isShopItem(obj) then return end
	local n = obj.Name:lower()
	for _, w in ipairs(weapons) do
		for _, kw in ipairs(w.keywords) do
			if n:find(kw, 1, true) then
				createBillboard(obj, w.display, w.color)
				return
			end
		end
	end
end
-- Lightweight instant rescan on drop (only players + child)
local function instantGunCheck()
	if not playerESPEnabled then return end
	for _, plr in ipairs(Players:GetPlayers()) do
		local char = plr.Character
		if char then
			for _, child in ipairs(char:GetChildren()) do
				tryTag(child)
			end
		end
		local bp = plr:FindFirstChild("Backpack")
		if bp then
			for _, child in ipairs(bp:GetChildren()) do
				tryTag(child)
			end
		end
	end
end
-- Instant drop detection + zero-delay force rescan
Workspace.DescendantAdded:Connect(function(child)
	local t = tick()
	if t - lastAddedTime < addedDebounce then return end
	lastAddedTime = t
	tryTag(child)
	if child:IsA("Model") or child:IsA("Folder") or child:IsA("Tool") then
		for _, sub in ipairs(child:GetDescendants()) do
			tryTag(sub)
		end
	end
	-- Zero delay: instant check when gun dropped
	instantGunCheck()
end)
Workspace.DescendantRemoving:Connect(function(obj)
	billboards[obj] = nil
end)
-- Distance updater
RunService.Heartbeat:Connect(function()
	local hrp = player.Character and player.Character:FindFirstChild("HumanoidRootPart")
	if not hrp then return end
	for item, data in pairs(billboards) do
		if not item.Parent or not data.part.Parent or not data.bb.Parent then
			billboards[item] = nil
			continue
		end
		if data.enabled then
			local studs = math.floor((hrp.Position - data.part.Position).Magnitude)
			data.dist.Text = studs .. " studs"
			data.dist.TextColor3 = studs < 300 and Color3.new(0,1,0) or studs < 1000 and Color3.new(1,1,0) or Color3.new(1,0.3,0.3)
		end
	end
end)
-- Periodic very fast scan (1 second) - only players (no lag)
task.spawn(function()
	while true do
		if playerESPEnabled then
			instantGunCheck()
		end
		if mapESPEnabled then
			-- Very light map check: only check recent changes via cache or skip heavy scan
			-- Full map only every 12s to avoid lag
			task.wait(12)
			if mapESPEnabled then
				for _, obj in ipairs(Workspace:GetChildren()) do -- only direct children, fast
					tryTag(obj)
				end
			end
		end
		task.wait(1)
	end
end)

-- targeting
local function isTargetValid(part)
	if not part or not part.Parent then return false end
	local plr = Players:GetPlayerFromCharacter(part.Parent)
	if isWhitelisted(plr) then return false end
	local hum = part.Parent:FindFirstChild("Humanoid")
	return hum and hum.Health > 0
end
local function getClosestTarget()
	local closest
	local shortest = fovRadius
	for _,plr in ipairs(Players:GetPlayers()) do
		if plr ~= player and plr.Character and not isWhitelisted(plr) then
			local char = plr.Character
			local hum = char:FindFirstChild("Humanoid")
			if hum and hum.Health > 0 then
				local effectivePartName = lockPartName
				if lockPartName == "Head" then
					if isTargetUnderLocalPlayer(char) then
						effectivePartName = "HumanoidRootPart"
					end
				end
				local part = char:FindFirstChild(effectivePartName)
				if part then
					local pos,onScreen = camera:WorldToViewportPoint(part.Position)
					if onScreen then
						local dist = (Vector2.new(pos.X,pos.Y) - Vector2.new(mouse.X,mouse.Y)).Magnitude
						if dist < shortest then
							shortest = dist
							closest = part
						end
					end
				end
			end
		end
	end
	return closest
end
local function hasLineOfSight(targetPart)
	if not targetPart then return false end
	local origin = camera.CFrame.Position
	local direction = targetPart.Position - origin
	local r = workspace:Raycast(origin,direction,LOSParams)
	if not r then return true end
	return r.Instance and r.Instance:IsDescendantOf(targetPart.Parent)
end

-- highlight
local function applyHighlight(char)
	clearHighlight()
	local h = Instance.new("Highlight")
	h.Adornee = char
	h.FillTransparency = 1
	h.OutlineTransparency = 0
	h.OutlineColor = Color3.fromRGB(170,0,255)
	h.Parent = char
	currentHighlight = h
	highlightTween = TweenService:Create( h, TweenInfo.new(0.7,Enum.EasingStyle.Sine,Enum.EasingDirection.InOut,-1,true), {OutlineColor = Color3.fromRGB(0,0,0)} )
	highlightTween:Play()
end

-- trigger
local function tryTrigger()
	if triggerDisabledByT then return end
	if os.clock() < triggerEnableTime then return end
	if not triggerBot then return end
	if not lockedPart then return end
	if not isTargetValid(lockedPart) then return end
	if not hasLineOfSight(lockedPart) then return end
	if os.clock() - lastShot < shotDelay then return end
	lastShot = os.clock()
	if mouse1click then
		mouse1click()
	else
		VirtualInputManager:SendMouseButtonEvent(0,0,0,true,game,0)
		task.wait()
		VirtualInputManager:SendMouseButtonEvent(0,0,0,false,game,0)
	end
end

RunService.RenderStepped:Connect(function()
	enforceSky()
	if rightClickLock then
		if not rightClickHeld then
			clearHighlight()
			return
		end
	else
		if not enabled then
			clearHighlight()
			return
		end
	end
	if not lockedPart then
		lockedPart = getClosestTarget()
	end
	if not isTargetValid(lockedPart) then
		if autoLock then
			lockedPart = getClosestTarget()
		else
			lockedPart = nil
			clearHighlight()
			return
		end
	end
	if not lockedPart then
		clearHighlight()
		return
	end
	local char = lockedPart.Parent
	if lockPartName == "Head" then
		local head = char:FindFirstChild("Head")
		local root = char:FindFirstChild("HumanoidRootPart")
		if head and root then
			local isUnder = isTargetUnderLocalPlayer(char)
			if isUnder and lockedPart == head then
				lockedPart = root
			elseif not isUnder and lockedPart == root then
				lockedPart = head
			end
		end
	end
	if char and (not currentHighlight or currentHighlight.Adornee ~= char) then
		applyHighlight(char)
	end
	camera.CFrame = CFrame.new(camera.CFrame.Position, lockedPart.Position)
	tryTrigger()
end)
